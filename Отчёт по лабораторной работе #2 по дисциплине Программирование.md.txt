       #include <iostream>
       #include <windows.h>
       
       using namespace std;
       
       int main()
       {
       	double x, a, b, z, e = 2.71828, dx, max, min, x_begin, x_end;
       
       	cin >> a >> b >> x_begin >> x_end >> dx;
       
       	for (x = x_begin; x <= x_end; x += dx) {
       		cout << x << "\t\t";
       		if (x <= a) {
       			z = pow(e, x);
       		}
       		else {
       			if (x > a and x < b) {
       				z = cos(x * x);
       			}
       			else {
       				if (x >= b) {
       					z = tan(x) + 8;
       				}
       			}
       		}
       		if (x == x_begin) {
       			min = z; max = z;
       		}
       		else {
       			if (z < min) min = z;
       			if (z > max) max = z;
       		}
       		cout << z << endl; 
       	}
       	cout << "min: " << min << "\t" << "max: " << max << endl;
       	system("pause"); 
       }
